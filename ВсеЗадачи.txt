int convert(char x)
{
    switch (x)
    {
    case 'I':
        return 1;
    case 'V':
        return 5;
    case 'X':
        return 10;
    case 'L':
        return 50;
    case 'C':
        return 100;
    case 'D':
        return 500;
    case 'M':
        return 1000;
    }
}

int main()
{
    string s;
    cin >> s;
    int sum = 0, c1 = 0;
    for (int i = s.length() - 1; i >= 0; i--)
    {
        int c = convert(s[i]);
        if (c >= c1)
            c1 = c;
        if (c == c1)
            sum += c;
        else
            sum -= c;
    }
    cout << sum;
}

#include <iostream>

using namespace std;

void perestanovka(int *a, int i, int j)
{
	int s = a[i];
	a[i] = a[j];
	a[j] = s;
}
bool perebor(int *a, int n)
{
	int i = n - 2;
	while (i != -1 && a[i] >= a[i + 1]) {
		i--;
	}

	if (i == -1) {
		return false;
	}

	int j = n - 1;
	while (a[i] >= a[j]) {
		j--;
	}

	perestanovka(a, i, j);
	int b = i + 1, c = n - 1;

	while (b < c) {
		perestanovka(a, b++, c--);
	}
}

int main() {
	setlocale(LC_ALL, "ru");

		a[i] = i + 1;
	}

		for (int i = 0; i < n; i++) {
			if (a[i] == i + 1) {
				count++;
				break;
			}
		}
	}
