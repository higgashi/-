#include <iostream>
#include <string>
using namespace std;

int convert(char x)
{
    switch (x)
    {
    case 'I':
        return 1;
    case 'V':
        return 5;
    case 'X':
        return 10;
    case 'L':
        return 50;
    case 'C':
        return 100;
    case 'D':
        return 500;
    case 'M':
        return 1000;
    }
}

int main()
{
    string s;
    cin >> s;
    int sum = 0, c1 = 0;
    for (int i = s.length() - 1; i >= 0; i--)
    {
        int c = convert(s[i]);
        if (c >= c1)
            c1 = c;
        if (c == c1)
            sum += c;
        else
            sum -= c;
    }
    cout << sum;
}

#include <iostream>
using namespace std;

int NOD_1(int a, int b)
{
    while (a > 0 && b > 0)

        if (a > b)
            a %= b;

        else
            b %= a;

    return a + b;
}

int NOD_2(int a, int b)
{
    while (a != b)

        if (a > b)
            a -= b;

        else
            b -= a;

    return a;
}

int main()
{
    int a, b;
    cin >> a >> b;
    cout << NOD_1(a, b) << endl << NOD_2(a, b);
